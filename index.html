function render() { const rnd = seededRandom(seedInput.value); const gridSize = 12; const canvasSize = canvas.width; const blockSize = canvasSize / (gridSize * 2); ctx.fillStyle = "black"; ctx.fillRect(0, 0, canvasSize, canvasSize); ctx.fillStyle = "white"; // 1. ПРАВИЛО ОКТАНТА const octantData = []; for (let r = 0; r < gridSize; r++) { octantData[r] = []; for (let c = 0; c <= r; c++) { octantData[r][c] = Math.floor(rnd() * SPRITES.length); } } // 2. РЕНДЕРИНГ СИММЕТРИИ D8 for (let r_cell = 0; r_cell < gridSize * 2; r_cell++) { for (let c_cell = 0; c_cell < gridSize * 2; c_cell++) { // Центрируем координаты: от -(gridSize) до +(gridSize-1) let y = r_cell - gridSize; let x = c_cell - gridSize; // Флаги отражения для квадрантов const flipV = y < 0; const flipH = x < 0; // Переводим в положительные индексы let abs_y = flipV ? -y - 1 : y; let abs_x = flipH ? -x - 1 : x; // Условие транспонирования для октанта (зеркало по диагонали) const trans = abs_x > abs_y; // Находим координаты в базовом треугольнике let s_r = trans ? abs_x : abs_y; let s_c = trans ? abs_y : abs_x; const patternIdx = octantData[s_r][s_c]; // Отрисовка const drawX = c_cell * blockSize; const drawY = r_cell * blockSize; // Важно: передаем flipV/flipH и trans для трансформации матрицы 7x7 drawTransformedSprite( ctx, patternIdx, drawX, drawY, blockSize, flipH, flipV, trans ); } } }
